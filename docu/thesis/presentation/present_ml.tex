      \begin{itemize}
        \small
        \item Math. formulation:
        \begin{equation*}
          \footnotesize
          \bm{z} = h(W \bm{x} + \bm{b})
        \end{equation*}
        \footnotesize
        with $\bm{x} \in \R^n$, $\bm{z} \in \R^m$, $\bm{b} \in \R^m$.
        \small
        \vspace{0.2cm}
        \item Amount of Params: 
        \footnotesize 
        $m \cdot n + m$
        \small
        \vspace{0.2cm}
        \item Amount of Operatios:
        \begin{equation*}
          \footnotesize
          \mathcal{T}(W \bm{x} + \bm{b}) \approx 2 (m \cdot n) + m
        \end{equation*}     
      \end{itemize}
